Program 1: Print Message in Different Lines

This program prints Hello word! using the printf() function in C. 

#include <stdio.h>

int main()
{
    printf("\n Hello word!");
    return 0;
}
Example Output:
Dice
Gadag Road
Koppal

Program 2: Read and Print Two Integer Numbers

This program reads two integer numbers from the user and prints them using the scanf() and printf() functions. 

#include <stdio.h>

int main()
{
    int a, b;

    printf("Enter the values of a and b: ");
    scanf("%d %d", &a, &b);

    printf("a = %d\nb = %d", a, b);

    return 0;
}
Example Output:
Input: a = 10, b = 20
Output:
a = 10
b = 20

Program 5: Add Two Numbers

This program reads two integer numbers and calculates their sum. The result is stored in a variable c. 

#include <stdio.h>

int main()
{
    int a, b, c;

    printf("Enter the values of a and b: ");
    scanf("%d %d", &a, &b);

    c = a + b;

    printf("a = %d\nb = %d\nc = %d", a, b, c);

    return 0;
}
Example Output:
Input: a = 10, b = 20
Output:
a = 10
b = 20
c = 30

Program 8: Divide Two Numbers

This program reads two integers from the user and divides the first number by the second. The result is stored in a float variable c to allow decimal values. 

#include <stdio.h>

int main()
{
    int a, b;
    float c;

    printf("Enter the values of a and b: ");
    scanf("%d %d", &a, &b);

    c = (float)a / b; 

    printf("a = %d\nb = %d\nc = %.2f", a, b, c);

    return 0;
}
Example Output:
Input: a = 10, b = 4
Output:
a = 10
b = 4
c = 2.50

Program 9: Calculating Average of Numbers

This program calculates the average of numbers. The user enters the total sum of the numbers, and the program divides it by 6 to find the average. 

#include <stdio.h>

int main()
{
    int total;
    float avg;

    printf("Enter the value for total: ");
    scanf("%d", &total);

    avg = (float)total / 6; 

    printf("Total = %d\nAverage = %.2f", total, avg);

    return 0;
}
Example Output:
Input: total = 54
Output:
Total = 54
Average = 9.00

Program 10: Calculate Amount

This program calculates the amount based on the rate and quantity entered by the user. The formula used is amount = rate × quantity. 

#include <stdio.h>

int main()
{
    int rate, qty, amount;

    printf("Enter the value for rate and quantity: ");
    scanf("%d %d", &rate, &qty);

    amount = rate * qty;

    printf("Rate = %d\nQuantity = %d\nAmount = %d", rate, qty, amount);

    return 0;
}
Example Output:
Input: rate = 50, qty = 4
Output:
Rate = 50
Quantity = 4
Amount = 200
